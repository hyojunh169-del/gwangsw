import streamlit as st import streamlit.components.v1 as components from pathlib import Path import base64 # 1. 스트림릿 페이지 기본 설정 st.set_page_config( page_title="뚱냥이 키우기 극악 버전", page_icon="🐱", layout="centered" ) # 2. 상단 타이틀 및 설명 st.title("🐱 뚱냥이 키우기 서비스") st.write("진화와 돌연사의 갈림길! 당신의 뚱냥이를 1톤(t) 이상의 '강승우'로 무사히 진화시켜보세요.") st.markdown("---") # 3. 프로젝트 폴더 구조에 따른 상대 경로 설정 html_file_path = Path(__file__).resolve().parent / "htmls" / "index.html" # 4. 파일 존재 여부 확인 후 안전 렌더링 if html_file_path.exists(): with open(html_file_path, "r", encoding="utf-8") as f: html_content = f.read() # [핵심 보정] 문자열 깨짐을 차단하기 위한 base64 안전 인코딩 변환 작업 b64_html = base64.b64encode(html_content.encode('utf-8')).decode('utf-8') components.html( f'', height=800, scrolling=False ) else: # 파일을 찾을 수 없을 때 출력하는 친절한 한국어 경고 메시지 st.error("앗! 게임 화면 파일(index.html)을 불러올 수 없습니다. 😿") st.warning(f"폴더 구조가 올바르게 배치되어 있는지 확인해 주세요.\n필요 경로: `{html_file_path}`")
